ColorHelper (C#)

C# 
namespace HnVue.UI.Themes;

/// <summary>
/// Helper class for accessing theme colors in code
/// </summary>
public static class ColorHelper
{
    /// <summary>
    /// Gets the primary color brush
    /// </summary>
    public static SolidColorBrush Primary => GetBrush("Primary.Main");

    /// <summary>
    /// Gets the error color brush
    /// </summary>
    public static SolidColorBrush Error => GetBrush("Semantic.Error");

    /// <summary>
    /// Gets the success color brush
    /// </summary>
    public static SolidColorBrush Success => GetBrush("Semantic.Success");

    /// <summary>
    /// Creates a color from hex string
    /// </summary>
    public static Color FromHex(string hex)
    {
        if (hex.StartsWith("#"))
            hex = hex.Substring(1);

        byte a = 255;
        byte r = byte.Parse(hex.Substring(0, 2), NumberStyles.HexNumber);
        byte g = byte.Parse(hex.Substring(2, 2), NumberStyles.HexNumber);
        byte b = byte.Parse(hex.Substring(4, 2), NumberStyles.HexNumber);

        if (hex.Length == 8)
            a = byte.Parse(hex.Substring(6, 2), NumberStyles.HexNumber);

        return Color.FromArgb(a, r, g, b);
    }

    private static SolidColorBrush GetBrush(string resourceKey)
    {
        return Application.Current.FindResource(resourceKey) as SolidColorBrush
            ?? throw new KeyNotFoundException($"Resource '{resourceKey}' not found.");
    }
}

ButtonHelper (C#)

C# 
namespace HnVue.UI.Controls;

/// <summary>
/// Attached properties for button customization
/// </summary>
public static class ButtonHelper
{
    /// <summary>
    /// Gets or sets the button size variant
    /// </summary>
    public static readonly DependencyProperty SizeProperty =
        DependencyProperty.RegisterAttached(
            "Size",
            typeof(ButtonSize),
            typeof(ButtonHelper),
            new PropertyMetadata(ButtonSize.Medium, OnSizeChanged));

    public static ButtonSize GetSize(DependencyObject obj) =>
        (ButtonSize)obj.GetValue(SizeProperty);

    public static void SetSize(DependencyObject obj, ButtonSize value) =>
        obj.SetValue(SizeProperty, value);

    /// <summary>
    /// Gets or sets whether the button is in loading state
    /// </summary>
    public static readonly DependencyProperty IsLoadingProperty =
        DependencyProperty.RegisterAttached(
            "IsLoading",
            typeof(bool),
            typeof(ButtonHelper),
            new PropertyMetadata(false, OnIsLoadingChanged));

    public static bool GetIsLoading(DependencyObject obj) =>
        (bool)obj.GetValue(IsLoadingProperty);

    public static void SetIsLoading(DependencyObject obj, bool value) =>
        obj.SetValue(IsLoadingProperty, value);

    private static void OnIsLoadingChanged(DependencyObject d, DependencyPropertyChangedEventArgs e)
    {
        if (d is Button button)
        {
            bool isLoading = (bool)e.NewValue;
            button.IsEnabled = !isLoading;
            button.Content = isLoading ? "Loading..." : button.Content;
        }
    }
}

public enum ButtonSize
{
    Small,   // 28px height
    Medium,  // 36px height
    Large    // 44px height
}

Converter Implementations (C#)

C# 
namespace HnVue.UI.Converters;

/// <summary>
/// Converts boolean values to visibility
/// </summary>
public class BoolToVisibilityConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        bool boolValue = value is bool b && b;
        bool invert = parameter?.ToString() == "Inverse";
        return (boolValue ^ invert) ? Visibility.Visible : Visibility.Collapsed;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        return value is Visibility visibility && visibility == Visibility.Visible;
    }
}

/// <summary>
/// Converts null values to visibility
/// </summary>
public class NullToVisibilityConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        bool invert = parameter?.ToString() == "Inverse";
        bool isNull = value == null;
        return (isNull ^ invert) ? Visibility.Collapsed : Visibility.Visible;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        throw new NotImplementedException();
    }
}

/// <summary>
/// Checks if string is empty or whitespace
/// </summary>
public class StringEmptyConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        bool isEmpty = string.IsNullOrWhiteSpace(value as string);
        bool invert = parameter?.ToString() == "Inverse";
        return isEmpty ^ invert;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        throw new NotImplementedException();
    }
}

Notification System

ToastService (C#)

C# 
namespace HnVue.UI.Services;

/// <summary>
/// Service for displaying toast notifications
/// </summary>
public interface IToastService
{
    /// <summary>
    /// Shows a toast notification with the specified message
    /// </summary>
    void Show(string message, ToastType type = ToastType.Info, TimeSpan? duration = null);

    /// <summary>
    /// Shows a success toast
    /// </summary>
    void ShowSuccess(string message);

    /// <summary>
    /// Shows an error toast
    /// </summary>
    void ShowError(string message);

    /// <summary>
    /// Shows a warning toast
    /// </summary>
    void ShowWarning(string message);
}

/// <summary>
/// Default toast service implementation
/// </summary>
public class ToastService : IToastService
{
    private readonly UIElement _parent;

    public ToastService(UIElement parent)
    {
        _parent = parent;
    }

    public void Show(string message, ToastType type = ToastType.Info, TimeSpan? duration = null)
    {
        Application.Current.Dispatcher.Invoke(() =>
        {
            var toast = new ToastControl
            {
                Message = message,
                Type = type
            };

            // Position and show toast
            var adorner = new ToastAdorner(_parent, toast);
            var layer = AdornerLayer.GetAdornerLayer(_parent);
            layer?.Add(adorner);

            // Auto-dismiss after duration
            var dismissDuration = duration ?? GetDefaultDuration(type);
            Task.Delay(dismissDuration).ContinueWith(_ =>
            {
                Application.Current.Dispatcher.Invoke(() =>
                {
                    layer?.Remove(adorner);
                });
            });
        });
    }

    public void ShowSuccess(string message) =>
        Show(message, ToastType.Success);

    public void ShowError(string message) =>
        Show(message, ToastType.Error);

    public void ShowWarning(string message) =>
        Show(message, ToastType.Warning);

    private TimeSpan GetDefaultDuration(ToastType type) => type switch
    {
        ToastType.Success => TimeSpan.FromSeconds(3),
        ToastType.Error => TimeSpan.FromSeconds(5),
        ToastType.Warning => TimeSpan.FromSeconds(4),
        _ => TimeSpan.FromSeconds(3)
    };
}

public enum ToastType
{
    Info,
    Success,
    Warning,
    Error
}

Extension Methods

UI Extensions (C#)

C# 
namespace HnVue.UI.Extensions;

/// <summary>
/// UI-related extension methods
/// </summary>
public static class UIExtensions
{
    /// <summary>
    /// Finds a child element by type
    /// </summary>
    public static T FindChild<T>(this DependencyObject parent) where T : DependencyObject
    {
        if (parent == null) return null;

        for (int i = 0; i < VisualTreeHelper.GetChildrenCount(parent); i++)
        {
            var child = VisualTreeHelper.GetChild(parent, i);

            if (child is T result)
                return result;

            var childResult = FindChild<T>(child);
            if (childResult != null)
                return childResult;
        }

        return null;
    }

    /// <summary>
    /// Finds a parent element by type
    /// </summary>
    public static T FindParent<T>(this DependencyObject child) where T : DependencyObject
    {
        while (child != null)
        {
            if (child is T parent)
                return parent;

            child = VisualTreeHelper.GetParent(child);
        }

        return null;
    }

    /// <summary>
    /// Gets all descendants of a specific type
    /// </summary>
    public static IEnumerable<T> GetDescendants<T>(this DependencyObject parent) where T : DependencyObject
    {
        if (parent == null) yield break;

        for (int i = 0; i < VisualTreeHelper.GetChildrenCount(parent); i++)
        {
            var child = VisualTreeHelper.GetChild(parent, i);

            if (child is T t)
                yield return t;

            foreach (var descendant in GetDescendants<T>(child))
                yield return descendant;
        }
    }

    /// <summary>
    /// Sets focus to element with delay (useful for async operations)
    /// </summary>
    public static async Task FocusAsync(this UIElement element)
    {
        await Task.Delay(50); // Allow UI to settle
        element.Focus();
        Keyboard.Focus(element);
    }
}

/// <summary>
/// Binding extensions for simplified MVVM
/// </summary>
public static class BindingExtensions
{
    /// <summary>
    /// Creates a one-way binding
    /// </summary>
    public static BindingBase OneWay<T>(Expression<Func<T>> propertyAccessor)
    {
        var propertyName = GetPropertyName(propertyAccessor);
        return new Binding(propertyName) { Mode = BindingMode.OneWay };
    }

    /// <summary>
    /// Creates a two-way binding
    /// </summary>
    public static BindingBase TwoWay<T>(Expression<Func<T>> propertyAccessor)
    {
        var propertyName = GetPropertyName(propertyAccessor);
        return new Binding(propertyName) { Mode = BindingMode.TwoWay, UpdateSourceTrigger = UpdateSourceTrigger.PropertyChanged };
    }

    private static string GetPropertyName<T>(Expression<Func<T>> propertyAccessor)
    {
        var memberExpression = propertyAccessor.Body as MemberExpression;
        return memberExpression?.Member.Name ?? string.Empty;
    }
}
