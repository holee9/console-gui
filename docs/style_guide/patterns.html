Navigation Patterns

Primary Navigation

Sidebar navigation with keyboard shortcuts.

XAML 
<!-- Sidebar Navigation -->
<UserControl x:Class="HnVue.UI.Components.NavigationSidebar">
    <ItemsControl ItemsSource="{Binding NavigationItems}">
        <ItemsControl.ItemTemplate>
            <DataTemplate>
                <Border Style="{StaticResource NavItem.Container}"
                        Command="{Binding NavigateCommand}">
                    <StackPanel Orientation="Horizontal" Spacing="12">
                        <TextBlock Text="{Binding Icon}"/>
                        <TextBlock Text="{Binding Label}"/>
                        <TextBlock Text="{Binding Shortcut}"
                                   Foreground="{StaticResource Neutral.TextMuted}"
                                   Margin="Auto,0,0,0"/>
                    </StackPanel>
                </Border>
            </DataTemplate>
        </ItemsControl.ItemTemplate>
    </ItemsControl>
</UserControl>

<!-- ViewModel with keyboard shortcuts -->
public class NavigationItem {
    public string Label { get; set; }
    public string Icon { get; set; }
    public string Shortcut { get; set; }
    public ICommand NavigateCommand { get; set; }
}

// Keyboard shortcut binding
var items = new List<NavigationItem> {
    new() { Label = "Worklist", Shortcut = "Alt+1", Icon = "≣" },
    new() { Label = "Studylist", Shortcut = "Alt+2", Icon = "📁" },
    new() { Label = "Acquisition", Shortcut = "Alt+3", Icon = "◎" },
    new() { Label = "Settings", Shortcut = "Alt+4", Icon = "⚙" }
};

    <Border Grid.Column="4" Grid.ColumnSpan="4">Column 2</Border>
    <Border Grid.Column="8" Grid.ColumnSpan="4">Column 3</Border>

    <Border Grid.Column="8" Grid.ColumnSpan="4">Sidebar</Border>

Single Column Form

Standard vertical form for data entry.

Two-Column Form

XAML Form Layout

XAML 
<!-- Form with Validation -->
<UserControl x:Class="HnVue.UI.Components.PatientRegistrationForm">
    <ScrollViewer VerticalScrollBarVisibility="Auto">
        <StackPanel Spacing="16" Margin="24">
            <TextBlock Text="Patient Registration"
                       Style="{StaticResource TextStyle.H1}"/>

            <!-- Required Fields -->
            <WrapPanel Orientation="Horizontal">
                <TextBox Width="300"
                         hw:TextField.HintText="Patient Name *"
                         Text="{Binding PatientName, UpdateSourceTrigger=PropertyChanged}"/>
                <TextBox Width="200"
                         hw:TextField.HintText="Patient ID"
                         Text="{Binding PatientId, UpdateSourceTrigger=PropertyChanged}"
                         IsReadOnly="{Binding IsPatientIdAutoGenerated}"/>
            </WrapPanel>

            <!-- Date of Birth -->
            <DatePicker SelectedDate="{Binding DateOfBirth}"
                        DisplayDateStart="{Binding MinDate}"
                        DisplayDateEnd="{Binding MaxDate}"/>

            <!-- Sex Selection -->
            <StackPanel Orientation="Horizontal" Spacing="16">
                <RadioButton Content="Male"
                             IsChecked="{Binding IsMale}"/>
                <RadioButton Content="Female"
                             IsChecked="{Binding IsFemale}"/>
            </StackPanel>

            <!-- Form Actions -->
            <StackPanel Orientation="Horizontal"
                        HorizontalAlignment="Right"
                        Spacing="12">
                <Button Content="Cancel"
                        Style="{StaticResource Button.Secondary}"
                        Command="{Binding CancelCommand}"/>
                <Button Content="Register"
                        Style="{StaticResource Button.Primary}"
                        Command="{Binding RegisterCommand}"/>
            </StackPanel>
        </StackPanel>
    </ScrollViewer>
</UserControl>

XAML State Patterns

XAML 
<!-- Loading State -->
<Grid Visibility="{Binding IsLoading, Converter={StaticResource BoolToVisibility}}">
    <StackPanel VerticalAlignment="Center" HorizontalAlignment="Center">
        <controls:ProgressRing IsActive="True"
                           Width="48"
                           Height="48"/>
        <TextBlock Text="Loading..."
                   HorizontalAlignment="Center"
                   Margin="0,12,0,0"/>
    </StackPanel>
</Grid>

<!-- Empty State -->
<Grid Visibility="{Binding IsEmpty, Converter={StaticResource BoolToVisibility}}">
    <StackPanel VerticalAlignment="Center"
                HorizontalAlignment="Center"
                Spacing="16">
        <TextBlock Text="📋"
                   FontSize="64"
                   HorizontalAlignment="Center"/>
        <TextBlock Text="No items found"
                   Style="{StaticResource TextStyle.H3}"
                   HorizontalAlignment="Center"/>
        <TextBlock Text="Try adjusting your filters"
                   Style="{StaticResource TextStyle.Body}"
                   Foreground="{StaticResource Neutral.TextMuted}"
                   HorizontalAlignment="Center"/>
        <Button Content="Clear Filters"
                 Command="{Binding ClearFiltersCommand}"/>
    </StackPanel>
</Grid>

<!-- Error State -->
<Grid Visibility="{Binding HasError, Converter={StaticResource BoolToVisibility}}">
    <StackPanel VerticalAlignment="Center"
                HorizontalAlignment="Center"
                Spacing="16">
        <TextBlock Text="⚠"
                   FontSize="64"
                   Foreground="{StaticResource Semantic.Error}"/>
        <TextBlock Text="{Binding ErrorMessage}"
                   Style="{StaticResource TextStyle.H3}"/>
        <StackPanel Orientation="Horizontal" Spacing="12">
            <Button Content="Retry"
                     Command="{Binding RetryCommand}"/>
            <Button Content="Cancel"
                     Style="{StaticResource Button.Secondary}"/>
        </StackPanel>
    </StackPanel>
</Grid>
